CREATE TABLE _ERPUploadsUserFile (
    IdSeq INT IDENTITY(1,1) PRIMARY KEY,
    UserId INT NULL,
    Filename NVARCHAR(1000) NULL,
    Type NVARCHAR(100) NULL,
    IsAvatar BIT NOT NULL DEFAULT 0,
    Path NVARCHAR(2000) NULL,
    Size INT NULL,
    IdxNo INT NULL,
    CreatedBy INT NULL,
    CreatedAt DATETIME2 DEFAULT GETDATE(),
    UpdatedBy INT NULL,
    UpdatedAt DATETIME2 DEFAULT GETDATE()
);
